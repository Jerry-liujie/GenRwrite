with avg_price as ( select i_category, avg(i_current_price) as avg_price from item group by i_category ), month_seq as ( select distinct (d_month_seq) from date_dim where d_year = 2002 and d_moy = 3 ) select a.ca_state state, count(*) cnt from customer_address a join customer c on a.ca_address_sk = c.c_current_addr_sk join store_sales s on c.c_customer_sk = s.ss_customer_sk join date_dim d on s.ss_sold_date_sk = d.d_date_sk join item i on s.ss_item_sk = i.i_item_sk join avg_price ap on i.i_category = ap.i_category where d.d_month_seq = (select * from month_seq) and i.i_current_price > 1.2 * ap.avg_price group by a.ca_state having count(*) >= 10 order by cnt, a.ca_state limit 100;
