with return_items as ( select i_item_id as item_id, sum(case when return_type = 'store' then return_quantity else 0 end) as sr_item_qty, sum(case when return_type = 'catalog' then return_quantity else 0 end) as cr_item_qty, sum(case when return_type = 'web' then return_quantity else 0 end) as wr_item_qty from ( select sr_item_sk as item_sk, sr_return_quantity as return_quantity, 'store' as return_type, sr_returned_date_sk as date_sk from store_returns union all select cr_item_sk, cr_return_quantity, 'catalog', cr_returned_date_sk from catalog_returns union all select wr_item_sk, wr_return_quantity, 'web', wr_returned_date_sk from web_returns ) returns join item on returns.item_sk = item.i_item_sk join date_dim on returns.date_sk = date_dim.d_date_sk and d_date in ( select d_date from date_dim where d_week_seq in ( select d_week_seq from date_dim where d_date in ('2001-07-13', '2001-09-10', '2001-11-16') ) ) group by i_item_id ) select item_id, sr_item_qty, sr_item_qty /(sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 sr_dev, cr_item_qty, cr_item_qty /(sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 cr_dev, wr_item_qty, wr_item_qty /(sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 * 100 wr_dev, (sr_item_qty + cr_item_qty + wr_item_qty) / 3.0 average from return_items order by item_id, sr_item_qty limit 100;
