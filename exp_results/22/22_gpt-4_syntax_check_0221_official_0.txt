select i.i_product_name, i.i_brand, i.i_class, i.i_category, avg(inv.inv_quantity_on_hand) qoh from (select inv.inv_item_sk, avg(inv.inv_quantity_on_hand) as inv_quantity_on_hand from inventory inv, date_dim d where inv.inv_date_sk = d.d_date_sk and d.d_month_seq between 1200 and 1200 + 11 group by inv.inv_item_sk) inv, item i where inv.inv_item_sk = i.i_item_sk group by rollup(i.i_product_name, i.i_brand, i.i_class, i.i_category) order by qoh, i.i_product_name, i.i_brand, i.i_class, i.i_category limit 100;
