select i_product_name, i_brand, i_class, i_category, avg(inv_quantity_on_hand) qoh from ( select inv_item_sk, avg(inv_quantity_on_hand) as inv_quantity_on_hand from inventory where inv_date_sk in ( select d_date_sk from date_dim where d_month_seq between 1200 and 1201 ) group by inv_item_sk ) inv, item i where inv.inv_item_sk = i.i_item_sk group by rollup( i_product_name, i_brand, i_class, i_category ) order by qoh, i_product_name, i_brand, i_class, i_category limit 100;
