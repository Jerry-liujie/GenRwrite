select i_item_id, i_item_desc, s_store_id, s_store_name, max(ss_net_profit) as store_sales_profit, max(sr_net_loss) as store_returns_loss, max(cs_net_profit) as catalog_sales_profit from ( select ss_item_sk, ss_store_sk, ss_customer_sk, ss_ticket_number, max(ss_net_profit) as ss_net_profit from store_sales, date_dim where d_moy = 4 and d_year = 1999 and d_date_sk = ss_sold_date_sk group by ss_item_sk, ss_store_sk, ss_customer_sk, ss_ticket_number ) as store_sales_agg, ( select sr_item_sk, sr_customer_sk, sr_ticket_number, max(sr_net_loss) as sr_net_loss from store_returns, date_dim where d_moy between 4 and 10 and d_year = 1999 and d_date_sk = sr_returned_date_sk group by sr_item_sk, sr_customer_sk, sr_ticket_number ) as store_returns_agg, ( select cs_item_sk, cs_bill_customer_sk, max(cs_net_profit) as cs_net_profit from catalog_sales, date_dim where d_moy between 4 and 10 and d_year = 1999 and d_date_sk = cs_sold_date_sk group by cs_item_sk, cs_bill_customer_sk ) as catalog_sales_agg, store, item where i_item_sk = store_sales_agg.ss_item_sk and s_store_sk = store_sales_agg.ss_store_sk and store_sales_agg.ss_customer_sk = store_returns_agg.sr_customer_sk and store_sales_agg.ss_item_sk = store_returns_agg.sr_item_sk and store_sales_agg.ss_ticket_number = store_returns_agg.sr_ticket_number and store_returns_agg.sr_customer_sk = catalog_sales_agg.cs_bill_customer_sk and store_returns_agg.sr_item_sk = catalog_sales_agg.cs_item_sk group by i_item_id, i_item_desc, s_store_id, s_store_name order by i_item_id, i_item_desc, s_store_id, s_store_name limit 100;
