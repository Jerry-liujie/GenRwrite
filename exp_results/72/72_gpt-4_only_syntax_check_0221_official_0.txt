select i_item_desc, w_warehouse_name, d1.d_week_seq, sum( case when cs_promo_sk is null then 1 else 0 end ) no_promo, sum( case when cs_promo_sk is not null then 1 else 0 end ) promo, count(*) total_cnt from ( select cs_item_sk, cs_promo_sk, cs_sold_date_sk, cs_ship_date_sk, cs_bill_cdemo_sk, cs_bill_hdemo_sk, cs_quantity, cs_order_number from catalog_sales where cs_sold_date_sk in ( select d_date_sk from date_dim where d_year = 1998 ) ) cs join inventory on (cs.cs_item_sk = inv_item_sk) join warehouse on (w_warehouse_sk = inv_warehouse_sk) join item on (i_item_sk = cs.cs_item_sk) join customer_demographics on (cs.cs_bill_cdemo_sk = cd_demo_sk and cd_marital_status = 'S') join household_demographics on (cs.cs_bill_hdemo_sk = hd_demo_sk and hd_buy_potential = '1001-5000') join date_dim d1 on (cs.cs_sold_date_sk = d1.d_date_sk) join date_dim d2 on (inv_date_sk = d2.d_date_sk) join date_dim d3 on (cs.cs_ship_date_sk = d3.d_date_sk) where d1.d_week_seq = d2.d_week_seq and inv_quantity_on_hand < cs.cs_quantity and d3.d_date > d1.d_date + 5 group by i_item_desc, w_warehouse_name, d1.d_week_seq order by total_cnt desc, i_item_desc, w_warehouse_name, d1.d_week_seq limit 100;
