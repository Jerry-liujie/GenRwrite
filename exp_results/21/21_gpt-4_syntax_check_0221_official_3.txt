with inv_data as ( select inv_item_sk, inv_warehouse_sk, sum( case when ( cast(d_date as date) < cast ('1999-06-22' as date) ) then inv_quantity_on_hand else 0 end ) as inv_before, sum( case when ( cast(d_date as date) >= cast ('1999-06-22' as date) ) then inv_quantity_on_hand else 0 end ) as inv_after from inventory join date_dim on inv_date_sk = d_date_sk where d_date between (cast ('1999-06-22' as date) - interval '30' day) and (cast ('1999-06-22' as date) + interval '30' day) group by inv_item_sk, inv_warehouse_sk ) select w_warehouse_name, i_item_id, inv_before, inv_after from inv_data join item on i_item_sk = inv_item_sk join warehouse on inv_warehouse_sk = w_warehouse_sk where i_current_price between 0.99 and 1.49 and ( case when inv_before > 0 then inv_after / inv_before else null end ) between 2.0 / 3.0 and 3.0 / 2.0 order by w_warehouse_name, i_item_id limit 100;
