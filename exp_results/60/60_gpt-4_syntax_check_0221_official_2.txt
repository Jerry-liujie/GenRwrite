with sales as ( select i_item_id, sum(case when ss.ss_ext_sales_price is not null then ss.ss_ext_sales_price when cs.cs_ext_sales_price is not null then cs.cs_ext_sales_price when ws.ws_ext_sales_price is not null then ws.ws_ext_sales_price else null end) as total_sales from item i left join store_sales ss on i.i_item_sk = ss.ss_item_sk left join catalog_sales cs on i.i_item_sk = cs.cs_item_sk left join web_sales ws on i.i_item_sk = ws.ws_item_sk left join date_dim d on ss.ss_sold_date_sk = d.d_date_sk or cs.cs_sold_date_sk = d.d_date_sk or ws.ws_sold_date_sk = d.d_date_sk left join customer_address ca on ss.ss_addr_sk = ca.ca_address_sk or cs.cs_bill_addr_sk = ca.ca_address_sk or ws.ws_bill_addr_sk = ca.ca_address_sk where i.i_category = 'Jewelry' and d.d_year = 2000 and d.d_moy = 10 and ca.ca_gmt_offset = -5 group by i.i_item_id ) select i_item_id, total_sales from sales where total_sales is not null order by i_item_id, total_sales limit 100;
