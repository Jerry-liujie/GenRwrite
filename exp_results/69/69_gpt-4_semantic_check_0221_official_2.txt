select cd_gender, cd_marital_status, cd_education_status, count(*) cnt1, cd_purchase_estimate, count(*) cnt2, cd_credit_rating, count(*) cnt3 from customer_demographics cd join ( select c.c_current_cdemo_sk, c.c_customer_sk from customer c join customer_address ca on c.c_current_addr_sk = ca.ca_address_sk where ca_state in ('IL', 'TX', 'ME') and not exists ( select * from web_sales ws join date_dim dd on ws.ws_sold_date_sk = dd.d_date_sk where c.c_customer_sk = ws.ws_bill_customer_sk and dd.d_year = 2002 and dd.d_moy between 1 and 3 ) and not exists ( select * from catalog_sales cs join date_dim dd on cs.cs_sold_date_sk = dd.d_date_sk where c.c_customer_sk = cs.cs_ship_customer_sk and dd.d_year = 2002 and dd.d_moy between 1 and 3 ) ) c on cd.cd_demo_sk = c.c_current_cdemo_sk where exists ( select * from store_sales ss join date_dim dd on ss.ss_sold_date_sk = dd.d_date_sk where c.c_customer_sk = ss.ss_customer_sk and dd.d_year = 2002 and dd.d_moy between 1 and 3 ) group by cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating order by cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating limit 100;
