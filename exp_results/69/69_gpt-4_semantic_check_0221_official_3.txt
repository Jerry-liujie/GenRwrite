with store_sales_customers as ( select distinct c_customer_sk from store_sales join date_dim on ss_sold_date_sk = d_date_sk where d_year = 2002 and d_moy between 1 and 1 + 2 ), web_sales_customers as ( select distinct ws_bill_customer_sk from web_sales join date_dim on ws_sold_date_sk = d_date_sk where d_year = 2002 and d_moy between 1 and 1 + 2 ), catalog_sales_customers as ( select distinct cs_ship_customer_sk from catalog_sales join date_dim on cs_sold_date_sk = d_date_sk where d_year = 2002 and d_moy between 1 and 1 + 2 ) select cd_gender, cd_marital_status, cd_education_status, count(*) cnt1, cd_purchase_estimate, count(*) cnt2, cd_credit_rating, count(*) cnt3 from customer c join customer_address ca on c.c_current_addr_sk = ca.ca_address_sk join customer_demographics cd on cd_demo_sk = c.c_current_cdemo_sk join store_sales_customers ssc on c.c_customer_sk = ssc.c_customer_sk where ca_state in ('IL', 'TX', 'ME') and not exists (select 1 from web_sales_customers where ws_bill_customer_sk = c.c_customer_sk) and not exists (select 1 from catalog_sales_customers where cs_ship_customer_sk = c.c_customer_sk) group by cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating order by cd_gender, cd_marital_status, cd_education_status, cd_purchase_estimate, cd_credit_rating limit 100;
