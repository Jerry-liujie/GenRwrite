with ssales as ( select c_last_name, c_first_name, s_store_name, i_color, sum(ss_net_paid) netpaid from store_sales join store_returns on ss_ticket_number = sr_ticket_number and ss_item_sk = sr_item_sk join item on ss_item_sk = i_item_sk join customer on ss_customer_sk = c_customer_sk join customer_address on c_current_addr_sk = ca_address_sk and c_birth_country <> upper(ca_country) join store on ss_store_sk = s_store_sk and s_zip = ca_zip and s_market_id = 5 where i_color in ('aquamarine', 'seashell') group by c_last_name, c_first_name, s_store_name, i_color ) select c_last_name, c_first_name, s_store_name, sum(netpaid) paid from ssales group by c_last_name, c_first_name, s_store_name, i_color having sum(netpaid) > ( select 0.05 * avg(netpaid) from ssales ) order by c_last_name, c_first_name, s_store_name;
