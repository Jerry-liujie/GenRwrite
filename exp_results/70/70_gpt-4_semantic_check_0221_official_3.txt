with top_states as ( select s_state, rank() over ( partition by s_state order by sum(ss_net_profit) desc ) as ranking from store_sales join store on s_store_sk = ss_store_sk join date_dim on d_date_sk = ss_sold_date_sk where d_month_seq between 1220 and 1220 + 11 group by s_state having ranking <= 5 ) select sum(ss_net_profit) as total_sum, s_state, s_county, grouping(s_state) + grouping(s_county) as lochierarchy, rank() over ( partition by grouping(s_state) + grouping(s_county), case when grouping(s_county) = 0 then s_state end order by sum(ss_net_profit) desc ) as rank_within_parent from store_sales join store on s_store_sk = ss_store_sk join date_dim d1 on d1.d_date_sk = ss_sold_date_sk join top_states on top_states.s_state = store.s_state where d1.d_month_seq between 1220 and 1220 + 11 group by rollup(s_state, s_county) order by lochierarchy desc, case when lochierarchy = 0 then s_state end, rank_within_parent limit 100;
