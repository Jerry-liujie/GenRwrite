with top_states as ( select store.s_state, rank() over ( partition by store.s_state order by sum(ss_net_profit) desc ) as ranking from store_sales join store on s_store_sk = ss_store_sk join date_dim on d_date_sk = ss_sold_date_sk where d_month_seq between 1220 and 1220 + 11 group by store.s_state ) select sum(ss_net_profit) as total_sum, store.s_state, s_county, grouping(store.s_state) + grouping(s_county) as lochierarchy, rank() over ( partition by grouping(store.s_state) + grouping(s_county), case when grouping(s_county) = 0 then store.s_state end order by sum(ss_net_profit) desc ) as rank_within_parent from store_sales join store on s_store_sk = ss_store_sk join date_dim d1 on d1.d_date_sk = ss_sold_date_sk join top_states on top_states.s_state = store.s_state where d1.d_month_seq between 1220 and 1220 + 11 and top_states.ranking <= 5 group by rollup(store.s_state, s_county) order by grouping(store.s_state) + grouping(s_county) desc, case when grouping(store.s_state) + grouping(s_county) = 0 then store.s_state end, rank_within_parent limit 100;
