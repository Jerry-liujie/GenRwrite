select i_item_id, i_item_desc, i_current_price from item join inventory on inv_item_sk = i_item_sk join date_dim on d_date_sk = inv_date_sk join store_sales on ss_item_sk = i_item_sk where i_current_price between 58 and 88 and d_date between '2001-01-13' and '2001-03-14' and i_manufact_id in (259, 559, 580, 485) and inv_quantity_on_hand between 100 and 500 group by i_item_id, i_item_desc, i_current_price order by i_item_id limit 100;
