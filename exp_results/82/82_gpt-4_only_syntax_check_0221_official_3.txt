select i_item_id, i_item_desc, i_current_price from item join ( select inv_item_sk, inv_date_sk from inventory where inv_quantity_on_hand between 100 and 500 ) as inv on inv.inv_item_sk = item.i_item_sk join ( select d_date_sk from date_dim where d_date between cast('2001-01-13' as date) and (cast('2001-01-13' as date) + interval '60' day) ) as dd on dd.d_date_sk = inv.inv_date_sk join store_sales on store_sales.ss_item_sk = item.i_item_sk where i_current_price between 58 and 58 + 30 and i_manufact_id in (259, 559, 580, 485) group by i_item_id, i_item_desc, i_current_price order by i_item_id limit 100;
