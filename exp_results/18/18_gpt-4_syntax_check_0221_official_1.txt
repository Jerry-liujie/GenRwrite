select i_item_id, ca_country, ca_state, ca_county, avg(cs_quantity) agg1, avg(cs_list_price) agg2, avg(cs_coupon_amt) agg3, avg(cs_sales_price) agg4, avg(cs_net_profit) agg5, avg(c_birth_year) agg6, avg(cd1.cd_dep_count) agg7 from catalog_sales cs join date_dim on cs.cs_sold_date_sk = date_dim.d_date_sk join item on cs.cs_item_sk = item.i_item_sk join customer_demographics cd1 on cs.cs_bill_cdemo_sk = cd1.cd_demo_sk and cd1.cd_gender = 'F' and cd1.cd_education_status = 'Primary' join customer c on cs.cs_bill_customer_sk = c.c_customer_sk and c.c_birth_month in (1, 3, 7, 11, 10, 4) join customer_demographics cd2 on c.c_current_cdemo_sk = cd2.cd_demo_sk join customer_address on c.c_current_addr_sk = customer_address.ca_address_sk where date_dim.d_year = 2001 and customer_address.ca_state in ('AL', 'MO', 'TN', 'GA', 'MT', 'IN', 'CA') group by rollup (i_item_id, ca_country, ca_state, ca_county) order by ca_country, ca_state, ca_county, i_item_id limit 100;
