select i_brand_id, i_brand, i_manufact_id, i_manufact, sum(ss_ext_sales_price) ext_price from ( select d_date_sk, ss_item_sk, ss_customer_sk, ss_store_sk, ss_ext_sales_price from date_dim, store_sales where d_date_sk = ss_sold_date_sk and d_moy = 11 and d_year = 2002 ) sales, ( select i_item_sk, i_brand_id, i_brand, i_manufact_id, i_manufact from item where i_manager_id = 14 ) item, ( select c_customer_sk, c_current_addr_sk from customer ) customer, ( select ca_address_sk, substr(ca_zip, 1, 5) ca_zip from customer_address ) address, ( select s_store_sk, substr(s_zip, 1, 5) s_zip from store ) store where sales.d_date_sk = item.i_item_sk and sales.ss_customer_sk = customer.c_customer_sk and customer.c_current_addr_sk = address.ca_address_sk and address.ca_zip <> store.s_zip and sales.ss_store_sk = store.s_store_sk group by i_brand, i_brand_id, i_manufact_id, i_manufact order by ext_price desc, i_brand, i_brand_id, i_manufact_id, i_manufact limit 100;
