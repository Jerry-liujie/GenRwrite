with sa as ( select ss_store_sk, ss_item_sk, sum(ss_sales_price) as revenue from store_sales join date_dim on ss_sold_date_sk = d_date_sk where d_month_seq between 1176 and 1176 + 11 group by ss_store_sk, ss_item_sk ), sb as ( select ss_store_sk, avg(revenue) as ave from sa group by ss_store_sk ), sc as ( select ss_store_sk, ss_item_sk, sum(ss_sales_price) as revenue from store_sales join date_dim on ss_sold_date_sk = d_date_sk where d_month_seq between 1176 and 1176 + 11 group by ss_store_sk, ss_item_sk ) select s_store_name, i_item_desc, sc.revenue, i_current_price, i_wholesale_cost, i_brand from store join sc on s_store_sk = sc.ss_store_sk join item on i_item_sk = sc.ss_item_sk join sb on sb.ss_store_sk = sc.ss_store_sk where sc.revenue <= 0.1 * sb.ave order by s_store_name, i_item_desc limit 100;
