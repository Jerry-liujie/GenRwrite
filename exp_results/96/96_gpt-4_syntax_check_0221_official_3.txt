select count(*) from (select ss_sold_time_sk, ss_hdemo_sk, ss_store_sk from store_sales where ss_store_sk = (select s_store_sk from store where s_store_name = 'ese')) as sales join (select hd_demo_sk from household_demographics where hd_dep_count = 0) as demographics on sales.ss_hdemo_sk = demographics.hd_demo_sk join (select t_time_sk from time_dim where t_hour = 8 and t_minute >= 30 or t_hour > 8) as time on sales.ss_sold_time_sk = time.t_time_sk limit 100;
