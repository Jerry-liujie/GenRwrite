with asceding as ( select *, rank() over ( order by rank_col asc ) rnk from ( select ss_item_sk as item_sk, avg(ss_net_profit) as rank_col from store_sales where ss_store_sk = 4 group by ss_item_sk having avg(ss_net_profit) > 0.9 *( select avg(ss_net_profit) from store_sales where ss_store_sk = 4 and ss_hdemo_sk is null group by ss_store_sk ) ) as subquery1 ), filtered_asceding as ( select * from asceding where rnk < 11 ), descending as ( select *, rank() over ( order by rank_col desc ) rnk from ( select ss_item_sk as item_sk, avg(ss_net_profit) as rank_col from store_sales where ss_store_sk = 4 group by ss_item_sk having avg(ss_net_profit) > 0.9 *( select avg(ss_net_profit) from store_sales where ss_store_sk = 4 and ss_hdemo_sk is null group by ss_store_sk ) ) as subquery2 ), filtered_descending as ( select * from descending where rnk < 11 ) select filtered_asceding.rnk, i1.i_product_name as best_performing, i2.i_product_name as worst_performing from filtered_asceding join item i1 on i1.i_item_sk = filtered_asceding.item_sk join filtered_descending on filtered_asceding.rnk = filtered_descending.rnk join item i2 on i2.i_item_sk = filtered_descending.item_sk order by filtered_asceding.rnk limit 100;
