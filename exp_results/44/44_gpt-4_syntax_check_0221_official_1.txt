with asceding as (select ss_item_sk item_sk, avg(ss_net_profit) rank_col, rank() over ( order by avg(ss_net_profit) asc) rnk from store_sales where ss_store_sk = 4 group by ss_item_sk having avg(ss_net_profit) > 0.9 * (select avg(ss_net_profit) from store_sales where ss_store_sk = 4 and ss_hdemo_sk is null )), filtered_asceding as (select * from asceding where rnk < 11), descending as (select ss_item_sk item_sk, avg(ss_net_profit) rank_col, rank() over ( order by avg(ss_net_profit) desc) rnk from store_sales where ss_store_sk = 4 group by ss_item_sk having avg(ss_net_profit) > 0.9 * (select avg(ss_net_profit) from store_sales where ss_store_sk = 4 and ss_hdemo_sk is null )), filtered_descending as (select * from descending where rnk < 11) select filtered_asceding.rnk, i1.i_product_name best_performing, i2.i_product_name worst_performing from filtered_asceding join item i1 on i1.i_item_sk = filtered_asceding.item_sk join filtered_descending on filtered_asceding.rnk = filtered_descending.rnk join item i2 on i2.i_item_sk = filtered_descending.item_sk order by filtered_asceding.rnk limit 100;
