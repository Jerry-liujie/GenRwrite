with ss as ( select i_manufact_id, sum(ss_ext_sales_price) as total_sales from item inner join store_sales on ss_item_sk = i_item_sk inner join date_dim on ss_sold_date_sk = d_date_sk and d_year = 2001 and d_moy = 3 inner join customer_address on ss_addr_sk = ca_address_sk and ca_gmt_offset = -5 where i_category = 'Books' group by i_manufact_id ), cs as ( select i_manufact_id, sum(cs_ext_sales_price) as total_sales from item inner join catalog_sales on cs_item_sk = i_item_sk inner join date_dim on cs_sold_date_sk = d_date_sk and d_year = 2001 and d_moy = 3 inner join customer_address on cs_bill_addr_sk = ca_address_sk and ca_gmt_offset = -5 where i_category = 'Books' group by i_manufact_id ), ws as ( select i_manufact_id, sum(ws_ext_sales_price) as total_sales from item inner join web_sales on ws_item_sk = i_item_sk inner join date_dim on ws_sold_date_sk = d_date_sk and d_year = 2001 and d_moy = 3 inner join customer_address on ws_bill_addr_sk = ca_address_sk and ca_gmt_offset = -5 where i_category = 'Books' group by i_manufact_id ) select i_manufact_id, sum(total_sales) as total_sales from ( select * from ss union all select * from cs union all select * from ws ) tmp group by i_manufact_id order by total_sales limit 100;
