with store_sales_year as ( select c_customer_id, c_first_name, c_last_name, c_email_address, d_year, sum(ss_ext_list_price - ss_ext_discount_amt) year_total from customer join store_sales on c_customer_sk = ss_customer_sk join date_dim on ss_sold_date_sk = d_date_sk where d_year in (1998, 1999) group by c_customer_id, c_first_name, c_last_name, c_email_address, d_year ), web_sales_year as ( select c_customer_id, d_year, sum(ws_ext_list_price - ws_ext_discount_amt) year_total from customer join web_sales on c_customer_sk = ws_bill_customer_sk join date_dim on ws_sold_date_sk = d_date_sk where d_year in (1998, 1999) group by c_customer_id, d_year ) select ss.customer_id, ss.customer_first_name, ss.customer_last_name, ss.customer_email_address from store_sales_year ss join web_sales_year ws on ss.customer_id = ws.customer_id and ss.d_year = ws.d_year join store_sales_year ss2 on ss.customer_id = ss2.customer_id and ss.d_year = ss2.d_year - 1 join web_sales_year ws2 on ss.customer_id = ws2.customer_id and ss.d_year = ws2.d_year - 1 where ss.d_year = 1999 and ss.year_total > 0 and ws.year_total > 0 and (ws.year_total / ws2.year_total) > (ss.year_total / ss2.year_total) order by ss.customer_id, ss.customer_first_name, ss.customer_last_name, ss.customer_email_address limit 100;
