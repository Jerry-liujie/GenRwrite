select substr(w_warehouse_name, 1, 20), sm_type, web_name, sum(is_30_days) as "30 days", sum(is_31_60_days) as "31-60 days", sum(is_61_90_days) as "61-90 days", sum(is_91_120_days) as "91-120 days", sum(is_over_120_days) as ">120 days" from ( select ws_warehouse_sk, ws_ship_mode_sk, ws_web_site_sk, (ws_ship_date_sk - ws_sold_date_sk <= 30)::int as is_30_days, ((ws_ship_date_sk - ws_sold_date_sk > 30) and (ws_ship_date_sk - ws_sold_date_sk <= 60))::int as is_31_60_days, ((ws_ship_date_sk - ws_sold_date_sk > 60) and (ws_ship_date_sk - ws_sold_date_sk <= 90))::int as is_61_90_days, ((ws_ship_date_sk - ws_sold_date_sk > 90) and (ws_ship_date_sk - ws_sold_date_sk <= 120))::int as is_91_120_days, (ws_ship_date_sk - ws_sold_date_sk > 120)::int as is_over_120_days from web_sales where ws_ship_date_sk in ( select d_date_sk from date_dim where d_month_seq between 1223 and 1234 ) ) ws_sub join warehouse w on ws_sub.ws_warehouse_sk = w.w_warehouse_sk join ship_mode sm on ws_sub.ws_ship_mode_sk = sm.sm_ship_mode_sk join web_site ws_site on ws_sub.ws_web_site_sk = ws_site.web_site_sk group by substr(w_warehouse_name, 1, 20), sm_type, web_name order by substr(w_warehouse_name, 1, 20), sm_type, web_name limit 100;
