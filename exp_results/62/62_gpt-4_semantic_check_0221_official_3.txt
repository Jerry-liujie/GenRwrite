select substr(w_warehouse_name, 1, 20), sm_type, web_name, sum(is_within_30_days) as "30 days", sum(is_within_60_days) as "31-60 days", sum(is_within_90_days) as "61-90 days", sum(is_within_120_days) as "91-120 days", sum(is_over_120_days) as ">120 days" from ( select ws_warehouse_sk, ws_ship_mode_sk, ws_web_site_sk, (ws_ship_date_sk - ws_sold_date_sk <= 30) as is_within_30_days, (ws_ship_date_sk - ws_sold_date_sk > 30 and ws_ship_date_sk - ws_sold_date_sk <= 60) as is_within_60_days, (ws_ship_date_sk - ws_sold_date_sk > 60 and ws_ship_date_sk - ws_sold_date_sk <= 90) as is_within_90_days, (ws_ship_date_sk - ws_sold_date_sk > 90 and ws_ship_date_sk - ws_sold_date_sk <= 120) as is_within_120_days, (ws_ship_date_sk - ws_sold_date_sk > 120) as is_over_120_days from web_sales where ws_ship_date_sk in ( select d_date_sk from date_dim where d_month_seq between 1223 and 1234 ) ) ws join warehouse w on ws.ws_warehouse_sk = w.w_warehouse_sk join ship_mode sm on ws.ws_ship_mode_sk = sm.sm_ship_mode_sk join web_site ws on ws.ws_web_site_sk = ws.web_site_sk group by substr(w.w_warehouse_name, 1, 20), sm.sm_type, ws.web_name order by substr(w.w_warehouse_name, 1, 20), sm.sm_type, ws.web_name limit 100;
