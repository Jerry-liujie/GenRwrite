with avg_discount as ( select ws_item_sk, avg(ws_ext_discount_amt) as avg_discount_amt from web_sales join date_dim on d_date_sk = ws_sold_date_sk where d_date between '2000-02-01' and (cast('2000-02-01' as date) + interval '90' day) group by ws_item_sk ) select sum(ws_ext_discount_amt) as "excess discount amount" from web_sales join item on i_item_sk = ws_item_sk join date_dim on d_date_sk = ws_sold_date_sk join avg_discount on avg_discount.ws_item_sk = item.i_item_sk where i_manufact_id = 714 and d_date between '2000-02-01' and (cast('2000-02-01' as date) + interval '90' day) and ws_ext_discount_amt > 1.3 * avg_discount.avg_discount_amt order by sum(ws_ext_discount_amt) limit 100;
