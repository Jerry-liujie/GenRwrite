select dt.d_year, item.i_category_id, item.i_category, sum(ss.ss_ext_sales_price) from date_dim dt join store_sales ss on dt.d_date_sk = ss.ss_sold_date_sk join item on ss.ss_item_sk = item.i_item_sk where item.i_manager_id = 1 and dt.d_moy = 11 and dt.d_year = 1998 group by dt.d_year, item.i_category_id, item.i_category order by sum(ss.ss_ext_sales_price) desc, dt.d_year, item.i_category_id, item.i_category limit 100;
